81    /// Check N0DES_SECRET_KEY environment variable for a valid API secret
82    pub fn api_secret_from_env(self) -> Result<Self> {
83        match std::env::var(API_SECRET_ENV_VAR_NAME) {
84            Ok(ticket_string) => {
85                let ticket = ApiSecret::from_str(&ticket_string)
86                    .context("invalid {API_SECRET_ENV_VAR_NAME}")?;
87                self.api_secret(ticket)
88            }
89            Err(VarError::NotPresent) => Err(anyhow!(
90                "{API_SECRET_ENV_VAR_NAME} environment variable is not set"
91            )),
92            Err(VarError::NotUnicode(e)) => Err(anyhow!(
93                "{API_SECRET_ENV_VAR_NAME} environment variable is not valid unicode: {:?}",
94                e
95            )),
96        }
97    }

99    /// set client API secret from an encoded string
100    pub fn api_secret_from_str(self, secret_key: &str) -> Result<Self> {
101        let key = ApiSecret::from_str(secret_key).context("invalid n0des api secret")?;
102        self.api_secret(key)
103    }

105    /// Use a shared secret & remote n0des endpoint ID contained within a ticket
106    /// to construct a n0des client. The resulting client will have "Client"
107    /// capabilities.
108    pub fn api_secret(mut self, ticket: ApiSecret) -> Result<Self> {
109        let local_id = self.endpoint.id();
110        let rcan = crate::caps::create_api_token_from_secret_key(
111            ticket.secret,
112            local_id,
113            self.cap_expiry,
114            Caps::for_shared_secret(),
115        )?;
116
117        self.remote = Some(ticket.remote);
118        self.rcan(rcan)
119    }

162    /// Create a new client, connected to the provide service node
163    #[must_use = "dropping the client will silently cancel all client tasks"]
164    pub async fn build(self) -> Result<Client, BuildError> {
165        debug!("starting iroh-n0des client");
166        let remote = self.remote.ok_or(BuildError::MissingRemote)?;
167        let capabilities = self.cap.ok_or(BuildError::MissingCapability)?;
168
169        let conn = IrohLazyRemoteConnection::new(self.endpoint, remote, ALPN.to_vec());
170        let client = N0desClient::boxed(conn);
171
172        let (tx, rx) = tokio::sync::mpsc::channel(1);
173        let metrics_task = AbortOnDropHandle::new(n0_future::task::spawn(
174            ClientActor {
175                capabilities,
176                client: client,
177                session_id: Uuid::new_v4(),
178                authorized: false,
179            }
180            .run(self.registry, self.metrics_interval, rx),
181        ));
182
183        Ok(Client {
184            message_channel: tx,
185            _actor_task: metrics_task,
186        })
187    }

190#[derive(thiserror::Error, Debug)]
191pub enum BuildError {
192    #[error("Missing remote endpoint to dial")]
193    MissingRemote,
194    #[error("Missing capability")]
195    MissingCapability,
196    #[error("Unauthorized")]
197    Unauthorized,
198    #[error("Remote error: {0}")]
199    Remote(#[from] RemoteError),
200    #[error("Rpc connection error: {0}")]
201    Rpc(irpc::Error),
202    #[error("Connection error: {0}")]
203    Connect(ConnectError),
204}
205
206impl From<irpc::Error> for BuildError {
207    fn from(value: irpc::Error) -> Self {
208        match value {
209            irpc::Error::Request {
210                source:
211                    irpc::RequestError::Connection {
212                        source: iroh::endpoint::ConnectionError::ApplicationClosed(frame),
213                        ..
214                    },
215                ..
216            } if frame.error_code == 401u32.into() => Self::Unauthorized,
217            value => Self::Rpc(value),
218        }
219    }
220}

271struct ClientActor {
272    capabilities: Rcan<Caps>,
273    client: N0desClient,
274    session_id: Uuid,
275    authorized: bool,
276}
277
278impl ClientActor {
279    async fn run(
280        mut self,
281        registry: Registry,
282        interval: Option<Duration>,
283        mut inbox: tokio::sync::mpsc::Receiver<ClientActorMessage>,
284    ) {
285        let registry = Arc::new(RwLock::new(registry));
286        let mut encoder = Encoder::new(registry);
287        let interval = interval.unwrap_or(Duration::from_secs(0));
288
289        let mut metrics_timer = n0_future::time::interval(interval);
290        loop {
291            tokio::select! {
292                biased;
293                Some(msg) = inbox.recv() => {
294                    match msg {
295                        ClientActorMessage::Ping{ done } => {
296                            if let Err(err) = self.send_ping(done).await {
297                                warn!("failed to send ping: {:#?}", err);
298                            }
299                        },
300                        ClientActorMessage::SendMetrics => {
301                            debug!("sending metrics manually triggered");
302                            if let Err(err) = self.send_metrics(&mut encoder).await {
303                                warn!("failed to push metrics: {:#?}", err);
304                            }
305                        }
306                    }
307                }
308                _ = metrics_timer.tick() => {
309                    trace!("metrics send tick");
310                    if let Err(err) = self.send_metrics(&mut encoder).await {
311                        warn!("failed to push metrics: {:#?}", err);
312                    }
313                },
314            }
315        }
316    }
317
318    // sends an authorization request to the server
319    async fn auth(&mut self) -> Result<()> {
320        if self.authorized {
321            return Ok(());
322        }
323        self.client
324            .rpc(Auth {
325                caps: self.capabilities.clone(),
326            })
327            .await
328            .inspect_err(|e| debug!("authorization failed: {:?}", e))?;
329        self.authorized = true;
330        Ok(())
331    }
332
333    async fn send_ping(&mut self, done: oneshot::Sender<()>) -> Result<()> {
334        self.auth().await?;
335        let req = rand::random();
336        self.client
337            .rpc(Ping { req })
338            .await
339            .inspect_err(|e| warn!("rpc ping error: {e}"))?;
340        let _ = done.send(());
341        Ok(())
342    }
343
344    async fn send_metrics(&mut self, encoder: &mut Encoder) -> Result<()> {
345        self.auth().await?;
346
347        let update = encoder.export();
348        let req = PutMetrics {
349            session_id: self.session_id,
350            update,
351        };
352
353        self.client
354            .rpc(req)
355            .await
356            .inspect_err(|e| warn!("rpc send metrics error: {e}"))?
357            .inspect_err(|e| warn!("metrics server response error: {e}"))?;
358        trace!("sent metrics");
359        Ok(())
360    }
361}

375    #[tokio::test]
376    #[temp_env_vars]
377    async fn test_api_key_from_env() {
378        // construct
379        let mut rng = rand::rng();
380        let shared_secret = SecretKey::generate(&mut rng);
381        let fake_endpoint_id = SecretKey::generate(&mut rng).public();
382        let n0des_ticket = ApiSecret::new(shared_secret.clone(), fake_endpoint_id);
383        unsafe {
384            std::env::set_var(API_SECRET_ENV_VAR_NAME, n0des_ticket.to_string());
385        };
386
387        let endpoint = Endpoint::empty_builder(iroh::RelayMode::Disabled)
388            .bind()
389            .await
390            .unwrap();
391
392        let builder = Client::builder(&endpoint).api_secret_from_env().unwrap();
393
394        let fake_endpoint_addr: EndpointAddr = fake_endpoint_id.into();
395        assert_eq!(builder.remote, Some(fake_endpoint_addr));
396
397        let rcan = crate::caps::create_api_token_from_secret_key(
398            shared_secret,
399            endpoint.id(),
400            builder.cap_expiry,
401            Caps::for_shared_secret(),
402        )
403        .unwrap();
404        assert_eq!(builder.cap, Some(rcan.clone()));
405        assert_eq!(rcan.capability(), &Caps::new([Cap::Client]));
406    }
