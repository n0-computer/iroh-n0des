41const DEFAULT_CAP_EXPIRY: Duration = Duration::from_secs(60 * 60 * 24 * 30); // 1 month
42const API_SECRET_ENV_VAR_NAME: &str = "N0DES_API_SECRET";

44impl ClientBuilder {
45    pub fn new(endpoint: &Endpoint) -> Self {
46        let mut registry = Registry::default();
47        registry.register_all(endpoint.metrics());
48
49        Self {
50            cap: None,
51            cap_expiry: DEFAULT_CAP_EXPIRY,
52            endpoint: endpoint.clone(),

81    /// Check N0DES_SECRET_KEY environment variable for a valid API secret
82    pub fn api_secret_from_env(self) -> Result<Self> {
83        match std::env::var(API_SECRET_ENV_VAR_NAME) {
84            Ok(ticket_string) => {
85                let ticket = ApiSecret::from_str(&ticket_string)
86                    .context("invalid {API_SECRET_ENV_VAR_NAME}")?;
87                self.api_secret(ticket)
88            }
89            Err(VarError::NotPresent) => Err(anyhow!(
90                "{API_SECRET_ENV_VAR_NAME} environment variable is not set"
91            )),
92            Err(VarError::NotUnicode(e)) => Err(anyhow!(
93                "{API_SECRET_ENV_VAR_NAME} environment variable is not valid unicode: {:?}",
94                e
95            )),
96        }
97    }
98
99    pub fn api_secret_from_str(self, secret_key: &str) -> Result<Self> {
100        let key = ApiSecret::from_str(secret_key).context("invalid n0des api secret")?;
101        self.api_secret(key)
102    }
103
104    /// Use a shared secret & remote n0des endpoint ID contained within a ticket
105    /// to construct a n0des client. The resulting client will have "Client"
106    /// capabilities.
107    pub fn api_secret(mut self, ticket: ApiSecret) -> Result<Self> {
108        let local_id = self.endpoint.id();
109        let rcan = crate::caps::create_api_token_from_secret_key(
110            ticket.secret,
111            local_id,
112            self.cap_expiry,
113            Caps::for_shared_secret(),
114        )?;
115
116        self.remote = Some(ticket.remote);
117        self.rcan(rcan)
118    }

120    /// Loads the private ssh key from the given path, and creates the needed capability.
121    #[cfg(feature = "ssh-key")]
122    pub async fn ssh_key_from_file<P: AsRef<std::path::Path>>(self, path: P) -> Result<Self> {
123        let file_content = tokio::fs::read_to_string(path).await?;
124        let private_key = ssh_key::PrivateKey::from_openssh(&file_content)?;
125
126        self.ssh_key(&private_key)

129    /// Creates the capability from the provided private ssh key.
130    #[cfg(feature = "ssh-key")]
131    pub fn ssh_key(mut self, key: &ssh_key::PrivateKey) -> Result<Self> {
132        let local_id = self.endpoint.id();
133        let rcan = crate::caps::create_api_token_from_ssh_key(
134            key,
135            local_id,
136            self.cap_expiry,
137            Caps::all(),
138        )?;
139        self.cap.replace(rcan);

141        Ok(self)
142    }

144    /// Sets the rcan directly.
145    pub fn rcan(mut self, cap: Rcan<Caps>) -> Result<Self> {
146        ensure!(
147            EndpointId::from_verifying_key(*cap.audience()) == self.endpoint.id(),
148            "invalid audience"
149        );
150        self.cap.replace(cap);
151        Ok(self)
152    }

161    /// Create a new client, connected to the provide service node
162    #[must_use = "dropping the client will silently cancel all client tasks"]
163    pub async fn build(self) -> Result<Client, BuildError> {
164        debug!("starting iroh-n0des client");
165        let remote = self.remote.ok_or(BuildError::MissingRemote)?;
166        let cap = self.cap.ok_or(BuildError::MissingCapability)?;
167
168        let conn = IrohLazyRemoteConnection::new(self.endpoint.clone(), remote, ALPN.to_vec());
169        let client = N0desClient::boxed(conn);
170
171        // If auth fails, the connection is aborted.
172        let () = client.rpc(Auth { caps: cap }).await?;
173
174        let (tx, rx) = tokio::sync::mpsc::channel(1);
175        let metrics_task = self.enable_metrics.map(|interval| {
176            debug!(interval = ?interval, "starting metrics task");
177            AbortOnDropHandle::new(n0_future::task::spawn(
178                MetricsTask {
179                    client: client.clone(),
180                    session_id: Uuid::new_v4(),
181                }
182                .run(self.registry, interval, rx),
183            ))
184        });
185
186        Ok(Client {
187            client,
188            metrics_channel: tx,
189            _metrics_task: metrics_task,
190        })
191    }

194#[derive(thiserror::Error, Debug)]
195pub enum BuildError {
196    #[error("Missing remote endpoint to dial")]
197    MissingRemote,
198    #[error("Missing capability")]
199    MissingCapability,
200    #[error("Unauthorized")]
201    Unauthorized,
202    #[error("Remote error: {0}")]
203    Remote(#[from] RemoteError),
204    #[error("Rpc connection error: {0}")]
205    Rpc(irpc::Error),
206    #[error("Connection error: {0}")]
207    Connect(ConnectError),
208}
209
210impl From<irpc::Error> for BuildError {
211    fn from(value: irpc::Error) -> Self {
212        match value {
213            irpc::Error::Request {
214                source:
215                    irpc::RequestError::Connection {
216                        source: iroh::endpoint::ConnectionError::ApplicationClosed(frame),
217                        ..
218                    },
219                ..
220            } if frame.error_code == 401u32.into() => Self::Unauthorized,
221            value => Self::Rpc(value),
222        }
223    }
224}

241    /// Pings the remote node.
242    pub async fn ping(&mut self) -> Result<(), Error> {
243        let req = rand::random();
244        let pong = self.client.rpc(Ping { req }).await?;
245        if pong.req == req {
246            Ok(())
247        } else {
248            Err(Error::Other(anyhow!("unexpected pong response")))
249        }
250    }

252    /// immideately send a single dump of metrics to n0des. It's not necessary
253    /// to call this function if you're using a non-zero metrics interval,
254    /// which will automatically propagate metrics on the set interval for you
255    pub async fn send_metrics(&self) -> Result<(), Error> {
256        self.metrics_channel
257            .send(())
258            .await
259            .map_err(|_| Error::Other(anyhow!("sending metrics")))
260    }

263struct MetricsTask {
264    client: N0desClient,
265    session_id: Uuid,
266}
267
268impl MetricsTask {
269    async fn run(
270        self,
271        registry: Registry,
272        interval: Duration,
273        mut trigger: tokio::sync::mpsc::Receiver<()>,
274    ) {
275        let registry = Arc::new(RwLock::new(registry));
276        let mut encoder = Encoder::new(registry);
277
278        let mut metrics_timer = n0_future::time::interval(interval);
279        loop {
280            tokio::select! {
281                biased;
282                _ = trigger.recv() => {
283                    trace!("metrics send tick");
284                    if let Err(err) = self.send_metrics(&mut encoder).await {
285                        warn!("failed to push metrics: {:#?}", err);
286                    }
287                }
288                _ = metrics_timer.tick() => {
289                    trace!("metrics send tick");
290                    if let Err(err) = self.send_metrics(&mut encoder).await {
291                        warn!("failed to push metrics: {:#?}", err);
292                    }
293                },
294            }
295        }
296    }
297
298    async fn send_metrics(&self, encoder: &mut Encoder) -> Result<()> {
299        let update = encoder.export();
300        let req = PutMetrics {
301            session_id: self.session_id,
302            update,
303        };
304
305        self.client
306            .rpc(req)
307            .await
308            .inspect_err(|e| warn!("rpc send metrics error: {e}"))?
309            .inspect_err(|e| warn!("metrics server response error: {e}"))?;
310        trace!("sent metrics");
311        Ok(())
312    }

327    #[tokio::test]
328    #[temp_env_vars]
329    async fn test_api_key_from_env() {
330        // construct
331        let mut rng = rand::rng();
332        let shared_secret = SecretKey::generate(&mut rng);
333        let fake_endpoint_id = SecretKey::generate(&mut rng).public();
334        let n0des_ticket = ApiSecret::new(shared_secret.clone(), fake_endpoint_id);
335        unsafe {
336            std::env::set_var(API_SECRET_ENV_VAR_NAME, n0des_ticket.to_string());
337        };
338
339        let endpoint = Endpoint::empty_builder(iroh::RelayMode::Disabled)
340            .bind()
341            .await
342            .unwrap();
343
344        let builder = Client::builder(&endpoint).api_secret_from_env().unwrap();
345
346        let fake_endpoint_addr: EndpointAddr = fake_endpoint_id.into();
347        assert_eq!(builder.remote, Some(fake_endpoint_addr));
348
349        let rcan = crate::caps::create_api_token_from_secret_key(
350            shared_secret,
351            endpoint.id(),
352            builder.cap_expiry,
353            Caps::for_shared_secret(),
354        )
355        .unwrap();
356        assert_eq!(builder.cap, Some(rcan.clone()));
357        assert_eq!(rcan.capability(), &Caps::new([Cap::Client]));
358    }
