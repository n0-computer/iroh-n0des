30/// Client is the main handle for interacting with n0des. It communicates with
31/// n0des entirely through an iroh endpoint, and is configured through a builder.
32/// Client requires either an Ssh Key or [`ApiSecret`]
33///
34/// ```no_run
35/// use iroh_n0des::Client;
36///
37/// async fn build_client() -> anyhow::Result<()> {
38///     let endpoint = iroh::Endpoint::bind().await?;
39///
40///     // needs N0DES_API_SECRET set to an environment variable
41///     // client will now push endpoint metrics to n0des.
42///     let client = Client::builder(&endpoint)
43///         .api_secret_from_str("MY_API_SECRET")?
44///         .build()
45///         .await;
46///
47///     Ok(())
48/// }
49/// ```
50///
51/// [`ApiSecret`]: crate::api_secret::ApiSecret
52#[derive(Debug)]
53pub struct Client {
54    message_channel: tokio::sync::mpsc::Sender<ClientActorMessage>,
55    _actor_task: AbortOnDropHandle<()>,
56}

58/// ClientBuilder provides configures and builds a n0des client, typically
59/// created with [`Client::builder`]
60pub struct ClientBuilder {
61    #[allow(dead_code)]
62    cap_expiry: Duration,
63    cap: Option<Rcan<Caps>>,
64    endpoint: Endpoint,
65    metrics_interval: Option<Duration>,

70const DEFAULT_CAP_EXPIRY: Duration = Duration::from_secs(60 * 60 * 24 * 30); // 1 month
71const API_SECRET_ENV_VAR_NAME: &str = "N0DES_API_SECRET";

73impl ClientBuilder {
74    pub fn new(endpoint: &Endpoint) -> Self {
75        let mut registry = Registry::default();
76        registry.register_all(endpoint.metrics());
77
78        Self {
79            cap: None,
80            cap_expiry: DEFAULT_CAP_EXPIRY,
81            endpoint: endpoint.clone(),
82            metrics_interval: Some(Duration::from_secs(10)),

110    /// Check N0DES_API_SECRET environment variable for a valid API secret
111    pub fn api_secret_from_env(self) -> Result<Self> {
112        match std::env::var(API_SECRET_ENV_VAR_NAME) {
113            Ok(ticket_string) => {
114                let ticket = ApiSecret::from_str(&ticket_string)
115                    .context("invalid {API_SECRET_ENV_VAR_NAME}")?;
116                self.api_secret(ticket)
117            }
118            Err(VarError::NotPresent) => Err(anyhow!(
119                "{API_SECRET_ENV_VAR_NAME} environment variable is not set"
120            )),
121            Err(VarError::NotUnicode(e)) => Err(anyhow!(
122                "{API_SECRET_ENV_VAR_NAME} environment variable is not valid unicode: {:?}",
123                e
124            )),
125        }
126    }
127
128    /// set client API secret from an encoded string
129    pub fn api_secret_from_str(self, secret_key: &str) -> Result<Self> {
130        let key = ApiSecret::from_str(secret_key).context("invalid n0des api secret")?;
131        self.api_secret(key)
132    }

134    /// Use a shared secret & remote n0des endpoint ID contained within a ticket
135    /// to construct a n0des client. The resulting client will have "Client"
136    /// capabilities.
137    ///
138    /// API secrets include remote details within them, and will set both the
139    /// remote and rcan values on the builder
140    pub fn api_secret(mut self, ticket: ApiSecret) -> Result<Self> {
141        let local_id = self.endpoint.id();
142        let rcan = crate::caps::create_api_token_from_secret_key(
143            ticket.secret,
144            local_id,
145            self.cap_expiry,
146            Caps::for_shared_secret(),
147        )?;
148
149        self.remote = Some(ticket.remote);
150        self.rcan(rcan)
151    }

153    /// Loads the private ssh key from the given path, and creates the needed capability.
154    #[cfg(feature = "ssh-key")]
155    pub async fn ssh_key_from_file<P: AsRef<std::path::Path>>(self, path: P) -> Result<Self> {
156        let file_content = tokio::fs::read_to_string(path).await?;
157        let private_key = ssh_key::PrivateKey::from_openssh(&file_content)?;
158
159        self.ssh_key(&private_key)
160    }
161
162    /// Creates the capability from the provided private ssh key.
163    #[cfg(feature = "ssh-key")]
164    pub fn ssh_key(mut self, key: &ssh_key::PrivateKey) -> Result<Self> {
165        let local_id = self.endpoint.id();
166        let rcan = crate::caps::create_api_token_from_ssh_key(
167            key,
168            local_id,
169            self.cap_expiry,
170            Caps::all(),
171        )?;
172        self.cap.replace(rcan);
173
174        Ok(self)
175    }
176
177    /// Sets the rcan directly.
178    pub fn rcan(mut self, cap: Rcan<Caps>) -> Result<Self> {
179        ensure!(
180            EndpointId::from_verifying_key(*cap.audience()) == self.endpoint.id(),
181            "invalid audience"
182        );
183        self.cap.replace(cap);
184        Ok(self)
185    }

187    /// Sets the remote to dial, must be provided either directly by calling
188    /// this method, or through calling the api_secret builder methods.
189    pub fn remote(mut self, remote: impl Into<EndpointAddr>) -> Self {
190        self.remote = Some(remote.into());
191        self
192    }

194    /// Create a new client, connected to the provide service node
195    #[must_use = "dropping the client will silently cancel all client tasks"]
196    pub async fn build(self) -> Result<Client, BuildError> {
197        debug!("starting iroh-n0des client");
198        let remote = self.remote.ok_or(BuildError::MissingRemote)?;
199        let capabilities = self.cap.ok_or(BuildError::MissingCapability)?;
200
201        let conn = IrohLazyRemoteConnection::new(self.endpoint, remote, ALPN.to_vec());
202        let client = N0desClient::boxed(conn);
203
204        let (tx, rx) = tokio::sync::mpsc::channel(1);
205        let metrics_task = AbortOnDropHandle::new(n0_future::task::spawn(
206            ClientActor {
207                capabilities,
208                client,
209                session_id: Uuid::new_v4(),
210                authorized: false,
211            }
212            .run(self.registry, self.metrics_interval, rx),
213        ));
214
215        Ok(Client {
216            message_channel: tx,
217            _actor_task: metrics_task,
218        })
219    }
220}

222#[derive(thiserror::Error, Debug)]
223pub enum BuildError {
224    #[error("Missing remote endpoint to dial")]
225    MissingRemote,
226    #[error("Missing capability")]
227    MissingCapability,
228    #[error("Unauthorized")]
229    Unauthorized,
230    #[error("Remote error: {0}")]
231    Remote(#[from] RemoteError),
232    #[error("Rpc connection error: {0}")]
233    Rpc(irpc::Error),
234    #[error("Connection error: {0}")]
235    Connect(ConnectError),
236}
237
238impl From<irpc::Error> for BuildError {
239    fn from(value: irpc::Error) -> Self {
240        match value {
241            irpc::Error::Request {
242                source:
243                    irpc::RequestError::Connection {
244                        source: iroh::endpoint::ConnectionError::ApplicationClosed(frame),
245                        ..
246                    },
247                ..
248            } if frame.error_code == 401u32.into() => Self::Unauthorized,
249            value => Self::Rpc(value),
250        }
251    }
252}

269    /// Pings the remote node.
270    pub async fn ping(&self) -> Result<Pong, Error> {
271        let (tx, rx) = oneshot::channel();
272        self.message_channel
273            .send(ClientActorMessage::Ping { done: tx })
274            .await
275            .map_err(|_| Error::Other(anyhow!("sending ping request")))?;
276
277        rx.await
278            .map_err(|e| Error::Other(anyhow!("response on internal channel: {:?}", e)))?
279            .map_err(Error::Remote)
280    }

282    /// immediately send a single dump of metrics to n0des. It's not necessary
283    /// to call this function if you're using a non-zero metrics interval,
284    /// which will automatically propagate metrics on the set interval for you
285    pub async fn push_metrics(&self) -> Result<(), Error> {
286        let (tx, rx) = oneshot::channel();
287        self.message_channel
288            .send(ClientActorMessage::SendMetrics { done: tx })
289            .await
290            .map_err(|_| Error::Other(anyhow!("sending metrics")))?;
291
292        rx.await
293            .map_err(|e| Error::Other(anyhow!("response on internal channel: {:?}", e)))?
294            .map_err(Error::Remote)
295    }

297    #[cfg(feature = "tickets")]
298    pub async fn publish_ticket<T: Ticket>(&self, name: String, ticket: T) -> Result<(), Error> {
299        let ticket_kind = T::KIND.to_string();
300        let ticket_bytes = ticket.to_bytes();
301        let (tx, rx) = oneshot::channel();
302
303        self.message_channel
304            .send(ClientActorMessage::PublishTicket {
305                name,
306                ticket_kind,
307                ticket_bytes,
308                done: tx,
309            })
310            .await
311            .map_err(|_| Error::Other(anyhow!("publishing ticket")))?;
312
313        rx.await
314            .map_err(|e| Error::Other(anyhow!("response on internal channel: {:?}", e)))?
315    }
316
317    #[cfg(feature = "tickets")]
318    pub async fn fetch_tickets<T: Ticket>(
319        &self,
320        offset: u32,
321        limit: u32,
322    ) -> Result<Vec<PublishedTicket<T>>, Error> {
323        let (tx, rx) = oneshot::channel();
324        self.message_channel
325            .send(ClientActorMessage::FetchTickets {
326                offset,
327                limit,
328                ticket_kind: T::KIND.to_string(),
329                done: tx,
330            })
331            .await
332            .map_err(|_| Error::Other(anyhow!("fetching tickets")))?;
333
334        let tickets = rx
335            .await
336            .map_err(|e| Error::Other(anyhow!("response on internal channel: {:?}", e)))??
337            .iter()
338            .map(|td| {
339                let ticket = T::from_bytes(&td.ticket_bytes).map_err(|e| {
340                    Error::Other(anyhow!("parsing n0des tickets list response: {:?}", e))
341                })?;
342                Ok::<PublishedTicket<T>, Error>(PublishedTicket {
343                    name: td.name.clone(),
344                    ticket,
345                })
346            })
347            .collect::<Result<Vec<PublishedTicket<T>>, _>>()?;
348
349        Ok(tickets)
350    }

382struct ClientActor {
383    capabilities: Rcan<Caps>,
384    client: N0desClient,
385    session_id: Uuid,
386    authorized: bool,
387}
388
389impl ClientActor {
390    async fn run(
391        mut self,
392        registry: Registry,
393        interval: Option<Duration>,
394        mut inbox: tokio::sync::mpsc::Receiver<ClientActorMessage>,
395    ) {
396        let registry = Arc::new(RwLock::new(registry));
397        let mut encoder = Encoder::new(registry);
398        let mut metrics_timer = interval.map(|interval| n0_future::time::interval(interval));
399        trace!("starting client actor");
400        loop {
401            trace!("client actor tick");
402            tokio::select! {
403                biased;
404                Some(msg) = inbox.recv() => {
405                    match msg {
406                        ClientActorMessage::Ping{ done } => {
407                            let res = self.send_ping().await;
408                            if let Err(err) = done.send(res) {
409                                debug!("failed to send ping: {:#?}", err);
410                                self.authorized = false;
411                            }
412                        },
413                        ClientActorMessage::SendMetrics{ done } => {
414                            trace!("sending metrics manually triggered");
415                            let res = self.send_metrics(&mut encoder).await;
416                            if let Err(err) = done.send(res) {
417                                debug!("failed to push metrics: {:#?}", err);
418                                self.authorized = false;
419                            }
420                        }
421                        #[cfg(feature = "tickets")]
422                        ClientActorMessage::PublishTicket{ name, ticket_kind, ticket_bytes, done } => {
423                            let res = self.tickets_publish(name, ticket_kind, ticket_bytes).await;
424                            if let Err(err) = done.send(res) {
425                                warn!("failed to create signal: {:#?}", err);
426                            }
427                        }
428                        #[cfg(feature = "tickets")]
429                        ClientActorMessage::FetchTickets{ done, ticket_kind, offset, limit } => {
430                            let res = self.tickets_fetch(ticket_kind, offset, limit).await;
431                            if let Err(err) = done.send(res) {
432                                warn!("failed to create signal: {:#?}", err);
433                            }
434                        }
435                    }
436                }
437                _ = async {
438                    if let Some(ref mut timer) = metrics_timer {
439                        timer.tick().await;
440                    } else {
441                        std::future::pending::<()>().await;
442                    }
443                } => {
444                    trace!("metrics send tick");
445                    if let Err(err) = self.send_metrics(&mut encoder).await {
446                        debug!("failed to push metrics: {:#?}", err);
447                        self.authorized = false;
448                    }
449                },
450            }
451        }
452    }

454    // sends an authorization request to the server
455    async fn auth(&mut self) -> Result<(), RemoteError> {
456        if self.authorized {
457            return Ok(());
458        }
459        trace!("client authorizing");
460        self.client
461            .rpc(Auth {
462                caps: self.capabilities.clone(),
463            })
464            .await
465            .inspect_err(|e| debug!("authorization failed: {:?}", e))
466            .map_err(|e| RemoteError::AuthError(e.to_string()))?;
467        self.authorized = true;
468        Ok(())
469    }
470
471    async fn send_ping(&mut self) -> Result<Pong, RemoteError> {
472        trace!("client actor send ping");
473        self.auth().await?;
474
475        let req = rand::random();
476        self.client
477            .rpc(Ping { req_id: req })
478            .await
479            .inspect_err(|e| warn!("rpc ping error: {e}"))
480            .map_err(|_| RemoteError::InternalServerError)
481    }
482
483    async fn send_metrics(&mut self, encoder: &mut Encoder) -> Result<(), RemoteError> {
484        trace!("client actor send metrics");
485        self.auth().await?;
486
487        let update = encoder.export();
488        // let delta = update_delta(&self.latest_ackd_update, &update);
489        let req = PutMetrics {
490            session_id: self.session_id,
491            update,
492        };
493
494        self.client
495            .rpc(req)
496            .await
497            .map_err(|_| RemoteError::InternalServerError)??;
498
499        Ok(())
500    }

502    #[cfg(feature = "tickets")]
503    async fn tickets_publish(
504        &self,
505        name: String,
506        ticket_kind: String,
507        ticket_bytes: Vec<u8>,
508    ) -> Result<(), Error> {
509        self.client
510            .rpc(PublishTicket {
511                name,
512                ticket_kind,
513                ticket: ticket_bytes,
514            })
515            .await?;
516        Ok(())
517    }

519    #[cfg(feature = "tickets")]
520    async fn tickets_fetch(
521        &self,
522        ticket_kind: String,
523        offset: u32,
524        limit: u32,
525    ) -> Result<Vec<TicketData>, Error> {
526        let req_id = rand::random();
527        let signals = self
528            .client
529            .rpc(ListTickets {
530                req_id,
531                ticket_kind,
532                offset,
533                limit,
534            })
535            .await?;
536        Ok(signals)
537    }

555        // construct
556        let mut rng = rand::rng();
557        let shared_secret = SecretKey::generate(&mut rng);
558        let fake_endpoint_id = SecretKey::generate(&mut rng).public();
559        let api_secret = ApiSecret::new(shared_secret.clone(), fake_endpoint_id);
560        unsafe {
561            std::env::set_var(API_SECRET_ENV_VAR_NAME, api_secret.to_string());
562        };
563
564        let endpoint = Endpoint::empty_builder(iroh::RelayMode::Disabled)
565            .bind()
566            .await
567            .unwrap();
568
569        let builder = Client::builder(&endpoint).api_secret_from_env().unwrap();
570
571        let fake_endpoint_addr: EndpointAddr = fake_endpoint_id.into();
572        assert_eq!(builder.remote, Some(fake_endpoint_addr));

574        let rcan = crate::caps::create_api_token_from_secret_key(
575            shared_secret,

577            builder.cap_expiry,
578            Caps::for_shared_secret(),
579        )
580        .unwrap();
581        assert_eq!(builder.cap, Some(rcan.clone()));
582        assert_eq!(rcan.capability(), &Caps::new([Cap::Client]));
