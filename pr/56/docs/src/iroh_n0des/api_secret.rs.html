11/// The secret material used to connect your n0des.iroh.computer project. The
12/// value of these should be treated like any other API key: guard them carefully.
13#[derive(Debug, Clone)]
14pub struct ApiSecret {
15    /// ED25519 secret used to construct rcans from
16    pub secret: SecretKey,
17    /// the n0des endpoint to direct requests to
18    pub remote: EndpointAddr,
19}
20
21impl Display for ApiSecret {
22    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
23        write!(f, "{}", Ticket::serialize(self))
24    }
25}
26
27#[derive(Serialize, Deserialize)]
28struct Variant0EndpointAddr {
29    endpoint_id: EndpointId,
30    addrs: BTreeSet<TransportAddr>,
31}
32
33/// Wire format for [`NodeTicket`].
34#[derive(Serialize, Deserialize)]
35enum TicketWireFormat {
36    Variant0(Variant0N0desTicket),
37}
38
39#[derive(Serialize, Deserialize)]
40struct Variant0N0desTicket {
41    secret: SecretKey,
42    addr: Variant0EndpointAddr,
43}
44
45impl Ticket for ApiSecret {
46    // KIND is the constant that's added to the front of a serialized ticket
47    // string. It should be a short, human readable string
48    const KIND: &'static str = "n0des";
49
50    fn to_bytes(&self) -> Vec<u8> {
51        let data = TicketWireFormat::Variant0(Variant0N0desTicket {
52            secret: self.secret.clone(),
53            addr: Variant0EndpointAddr {
54                endpoint_id: self.remote.id,
55                addrs: self.remote.addrs.clone(),
56            },
57        });
58        postcard::to_stdvec(&data).expect("postcard serialization failed")
59    }
60
61    fn from_bytes(bytes: &[u8]) -> Result<Self, ParseError> {
62        let res: TicketWireFormat = postcard::from_bytes(bytes)?;
63        let TicketWireFormat::Variant0(Variant0N0desTicket { secret, addr }) = res;
64        Ok(Self {
65            secret,
66            remote: EndpointAddr {
67                id: addr.endpoint_id,
68                addrs: addr.addrs.clone(),
69            },
70        })
71    }
72}
73
74impl FromStr for ApiSecret {
75    type Err = ParseError;
76
77    fn from_str(s: &str) -> Result<Self, Self::Err> {
78        iroh_tickets::Ticket::deserialize(s)
79    }
80}

82impl ApiSecret {
83    /// Creates a new ticket.
84    pub fn new(secret: SecretKey, remote: impl Into<EndpointAddr>) -> Self {
85        Self {
86            secret,
87            remote: remote.into(),
88        }
89    }
90
91    /// The [`EndpointAddr`] of the provider for this ticket.
92    pub fn addr(&self) -> &EndpointAddr {
93        &self.remote
94    }
