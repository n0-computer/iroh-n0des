504#[derive(Default)]
505struct LocalActor {
506    traces_by_name: BTreeMap<String, Uuid>,
507    traces: BTreeMap<Uuid, TraceState>,
508}
509
510struct TraceState {
511    init: InitTrace,
512    nodes: Vec<EndpointInfoWithAddr>,
513    barrier_start: Vec<oneshot::Sender<RemoteResult<WaitStartResponse>>>,
514    barrier_checkpoint: BTreeMap<CheckpointId, Vec<oneshot::Sender<RemoteResult<()>>>>,
515}
516
517impl TraceState {
518    fn node_count(&self) -> usize {
519        self.init.info.node_count as usize
520    }
521}
522impl LocalActor {
523    pub fn spawn(rx: tokio::sync::mpsc::Receiver<TraceMessage>) {
524        let actor = Self::default();
525        tokio::task::spawn(actor.run(rx));
526    }
527
528    pub async fn run(mut self, mut rx: tokio::sync::mpsc::Receiver<TraceMessage>) {
529        while let Some(message) = rx.recv().await {
530            self.handle_message(message).await;
531        }
532    }
533    async fn handle_message(&mut self, message: TraceMessage) {
534        match message {
535            TraceMessage::GetSession(_msg) => {}
536            TraceMessage::InitTrace(msg) => {
537                let WithChannels { inner, tx, .. } = msg;
538                if self.traces_by_name.contains_key(&inner.info.name) {
539                    return send_err(tx, RemoteError::other("Trace already initialized")).await;
540                }
541                let uuid = Uuid::now_v7();
542                self.traces_by_name.insert(inner.info.name.clone(), uuid);
543                self.traces.insert(
544                    uuid,
545                    TraceState {
546                        init: inner,
547                        nodes: Default::default(),
548                        barrier_start: Default::default(),
549                        barrier_checkpoint: Default::default(),
550                    },
551                );
552                tx.send(Ok(uuid)).await.ok();
553            }
554            TraceMessage::GetTrace(msg) => {
555                let WithChannels { inner, tx, .. } = msg;
556                let GetTrace {
557                    session_id: _,
558                    name,
559                } = inner;
560                let Some((trace_id, info)) = self
561                    .traces_by_name
562                    .get(&name)
563                    .and_then(|trace_id| self.traces.get_key_value(trace_id))
564                else {
565                    return send_err(tx, RemoteError::other("Trace not initialized")).await;
566                };
567                tx.send(Ok(GetTraceResponse {
568                    trace_id: *trace_id,
569                    info: info.init.info.clone(),
570                    setup_data: info.init.setup_data.clone(),
571                }))
572                .await
573                .ok();
574            }
575            TraceMessage::StartNode(msg) => {
576                let WithChannels { inner, tx, .. } = msg;
577                if self.traces.contains_key(&inner.trace_id) {
578                    tx.send(Ok(StartNodeResponse {})).await.ok();
579                } else {
580                    send_err(tx, RemoteError::other("Trace not initialized")).await;
581                }
582            }
583            TraceMessage::EndNode(msg) => {
584                // noop
585                msg.tx.send(Ok(())).await.ok();
586            }
587            TraceMessage::CloseTrace(msg) => {
588                // noop
589                msg.tx.send(Ok(())).await.ok();
590            }
591            TraceMessage::PutCheckpoint(msg) => {
592                // noop
593                msg.tx.send(Ok(())).await.ok();
594            }
595            TraceMessage::PutLogs(msg) => {
596                // noop
597                msg.tx.send(Ok(())).await.ok();
598            }
599            TraceMessage::PutMetrics(msg) => {
600                // noop
601                msg.tx.send(Ok(())).await.ok();
602            }
603            TraceMessage::WaitCheckpoint(msg) => {
604                let WithChannels { inner, tx, .. } = msg;
605                let Some(trace) = self.traces.get_mut(&inner.trace_id) else {
606                    return send_err(tx, RemoteError::TraceNotFound).await;
607                };
608                let node_count = trace.node_count();
609                let barrier = trace
610                    .barrier_checkpoint
611                    .entry(inner.checkpoint_id)
612                    .or_default();
613                barrier.push(tx);
614                debug!(trace_id=%inner.trace_id, checkpoint=inner.checkpoint_id, count=barrier.len(), total=node_count, "wait checkpoint");
615                if barrier.len() == node_count {
616                    debug!(trace_id=%inner.trace_id, checkpoint=inner.checkpoint_id, "release");
617                    barrier
618                        .drain(..)
619                        .map(|tx| tx.send(Ok(())))
620                        .into_unordered_stream()
621                        .count()
622                        .await;
623                }
624            }
625            TraceMessage::WaitStart(msg) => {
626                let WithChannels { inner, tx, .. } = msg;
627                let Some(trace) = self.traces.get_mut(&inner.trace_id) else {
628                    return send_err(tx, RemoteError::TraceNotFound).await;
629                };
630                trace.nodes.push(inner.info);
631                trace.barrier_start.push(tx);
632                if trace.barrier_start.len() == trace.init.info.node_count as usize {
633                    let data = WaitStartResponse {
634                        infos: trace.nodes.clone(),
635                    };
636                    for tx in trace.barrier_start.drain(..) {
637                        tx.send(Ok(data.clone())).await.ok();
638                    }
639                }
640            }
641        }
642    }
643}
