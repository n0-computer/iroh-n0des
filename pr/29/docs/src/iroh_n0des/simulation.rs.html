58/// Helper trait for async functions.
59///
60/// This is needed because with a simple `impl Fn() -> Fut`, we can't
61/// express a variadic lifetime bound from the future to the function parameter.
62/// `impl AsyncFn` would allow this, but that doesn't allow to express a `Send`
63/// bound on the future.
64pub trait AsyncCallback<'a, A1: 'a, A2: 'a, T: 'a>:
65    'static + Send + Sync + Fn(&'a mut A1, &'a A2) -> Self::Fut
66{
67    type Fut: Future<Output = T> + Send;
68}
69
70impl<'a, A1: 'a, A2: 'a, T: 'a, Out, F> AsyncCallback<'a, A1, A2, T> for F
71where
72    Out: Send + Future<Output = T>,
73    F: 'static + Sync + Send + Fn(&'a mut A1, &'a A2) -> Out,
74{
75    type Fut = Out;
76}

208/// Trait for types that can be spawned as simulation nodes.
209///
210/// This trait is generic over `D: SetupData`, which is the type returned from the
211/// user-defined setup function (see [`Builder::with_setup`]). If not using the setup
212/// step, `D` defaults to the unit type `()`.
213///
214/// Implement this trait on your node type to be able to spawn the node in a simulation
215/// context. The only required method is [`Spawn::spawn`], which must return your spawned node.
216pub trait Spawn<D: SetupData = ()>: Node + 'static {
217    /// Spawns a new instance of this node type.
218    ///
219    /// # Errors
220    ///
221    /// Returns an error if the node fails to initialize properly.
222    fn spawn(context: &mut SpawnContext<'_, D>) -> impl Future<Output = Result<Self>> + Send
223    where
224        Self: Sized;
225
226    /// Spawns a new instance as a dynamically-typed node.
227    ///
228    /// This calls `spawn` and boxes the result.
229    ///
230    /// # Errors
231    ///
232    /// Returns an error if the node fails to initialize properly.
233    fn spawn_dyn<'a>(context: &'a mut SpawnContext<'a, D>) -> BoxFuture<'a, Result<BoxNode>>
234    where
235        Self: Sized,
236    {
237        Box::pin(async move {
238            let node = Self::spawn(context).await?;
239            let node: Box<dyn DynNode> = Box::new(node);
240            anyhow::Ok(node)
241        })
242    }
243
244    /// Creates a new builder for this node type with the given round function.
245    ///
246    /// The round function will be called each simulation round and should return
247    /// `Ok(true)` to continue or `Ok(false)` to stop early.
248    fn builder(
249        round_fn: impl for<'a> AsyncCallback<'a, Self, RoundContext<'a, D>, Result<bool>>,
250    ) -> NodeBuilder<Self, D>
251    where
252        Self: Sized,
253    {
254        NodeBuilder::new(round_fn)
255    }
256}

377impl<N: Spawn<D>, D: SetupData> NodeBuilder<N, D> {
378    /// Creates a new node builder with the given round function.
379    ///
380    /// The round function will be called each simulation round and should return
381    /// `Ok(true)` to continue or `Ok(false)` to stop early.
382    pub fn new(
383        round_fn: impl for<'a> AsyncCallback<'a, N, RoundContext<'a, D>, Result<bool>>,
384    ) -> Self {
385        let spawn_fn: BoxedSpawnFn<D> = Arc::new(N::spawn_dyn);
386        let round_fn: BoxedRoundFn<D> = Arc::new(move |node, context| {
387            let node = node
388                .as_any_mut()
389                .downcast_mut::<N>()
390                .expect("unreachable: type is statically guaranteed");
391            Box::pin(round_fn(node, context))
392        });
393        Self {
394            phantom: PhantomData,
395            spawn_fn,
396            round_fn,
397            check_fn: None,
398        }
399    }
400
401    /// Adds a check function that will be called after each round.
402    ///
403    /// The check function can verify node state and return an error to fail
404    /// the simulation if invariants are violated.
405    ///
406    /// # Errors
407    ///
408    /// The check function should return an error if validation fails.
409    pub fn check(
410        mut self,
411        check_fn: impl 'static + for<'a> Fn(&'a N, &RoundContext<'a, D>) -> Result<()>,
412    ) -> Self {
413        let check_fn: BoxedCheckFn<D> = Arc::new(move |node, context| {
414            let node = node
415                .as_any()
416                .downcast_ref::<N>()
417                .expect("unreachable: type is statically guaranteed");
418            check_fn(node, context)
419        });
420        self.check_fn = Some(check_fn);
421        self
422    }
423
424    fn erase(self) -> ErasedNodeBuilder<D> {
425        ErasedNodeBuilder {
426            spawn_fn: self.spawn_fn,
427            round_fn: self.round_fn,
428            check_fn: self.check_fn,
429        }
430    }
431}

436    idx: u32,
437    round_fn: BoxedRoundFn<D>,
438    check_fn: Option<BoxedCheckFn<D>>,

445impl<D: SetupData> SimNode<D> {
446    async fn spawn_and_run(
447        builder: ErasedNodeBuilder<D>,
448        client: TraceClient,
449        trace_id: Uuid,
450        node_index: u32,
451        setup_data: &D,
452        rounds: u32,
453    ) -> Result<()> {
454        let secret_key = SecretKey::generate(&mut rand::rngs::OsRng);
455        let info = NodeInfo {
456            // TODO: Only assign node id if endpoint was created.
457            node_id: Some(secret_key.public()),
458            idx: node_index,
459            label: None,
460        };
461        let mut registry = Registry::default();
462        let mut context = SpawnContext {
463            setup_data,
464            node_index,
465            secret_key,
466            registry: &mut registry,
467        };
468        let node = (builder.spawn_fn)(&mut context).await?;
469
470        // TODO(Frando): enable metrics support for iroh 0.35
471        // iroh@0.35 uses iroh-metrics@0.34, which is a major version down from iroh-metrics@0.35
472        // which is used here, thus the traits are incompatible.
473        // Add shim to iroh-metrics@0.35 to support registering metrics from iroh-metrics@0.34
474        #[cfg(not(feature = "iroh_v035"))]
475        if let Some(endpoint) = node.endpoint() {
476            registry.register_all(endpoint.metrics());
477        }
478
479        let mut node = Self {
480            node,
481            trace_id,
482            idx: node_index,
483            info,
484            round: 0,
485            round_fn: builder.round_fn,
486            check_fn: builder.check_fn,
487            metrics_encoder: Encoder::new(Arc::new(RwLock::new(registry))),
488            all_nodes: Default::default(),
489        };
490
491        let res = node.run(&client, setup_data, rounds).await;
492        if let Err(err) = &res {
493            warn!("node failed: {err:#}");
494        }
495        res
496    }
497
498    async fn run(&mut self, client: &TraceClient, setup_data: &D, rounds: u32) -> Result<()> {
499        let client = client.start_node(self.trace_id, self.info.clone()).await?;
500
501        info!(idx = self.idx, "start");
502
503        let info = NodeInfoWithAddr {
504            addr: self.my_addr().await,
505            info: self.info.clone(),
506        };
507        let all_nodes = client.wait_start(info).await?;
508        self.all_nodes = all_nodes;
509
510        let result = self.run_rounds(&client, setup_data, rounds).await;
511
512        if let Err(err) = self.node.shutdown().await {
513            warn!("failure during node shutdown: {err:#}");
514        }
515
516        client.end(to_str_err(&result)).await?;
517
518        result
519    }
520
521    async fn run_rounds(
522        &mut self,
523        client: &ActiveTrace,
524        setup_data: &D,
525        rounds: u32,
526    ) -> Result<()> {
527        while self.round < rounds {
528            if !self
529                .run_round(client, setup_data)
530                .await
531                .with_context(|| format!("failed at round {}", self.round))?
532            {
533                return Ok(());
534            }
535            self.round += 1;
536        }
537        Ok(())
538    }
539
540    #[tracing::instrument(name="round", skip_all, fields(round=self.round))]
541    async fn run_round(&mut self, client: &ActiveTrace, setup_data: &D) -> Result<bool> {
542        info!("start round");
543        let context = RoundContext {
544            round: self.round,
545            node_index: self.idx,
546            setup_data,
547            all_nodes: &self.all_nodes,
548        };
549
550        let result = (self.round_fn)(&mut self.node, &context).await;
551
552        let checkpoint = (context.round + 1) as u64;
553        let label = format!("Round {} end", context.round);
554        client
555            .put_checkpoint(checkpoint, Some(label), to_str_err(&result))
556            .await?;
557
558        // TODO(Frando): Couple metrics to node idx, not node id.
559        if let Some(node_id) = self.node_id() {
560            client
561                .put_metrics(node_id, Some(checkpoint), self.metrics_encoder.export())
562                .await?;
563        }
564
565        client.wait_checkpoint(checkpoint).await?;
566
567        if let Some(check_fn) = self.check_fn.as_ref() {
568            (check_fn)(&self.node, &context).with_context(|| "check function failed")?;
569        }
570
571        info!("end round");
572
573        result
574    }
575
576    fn node_id(&self) -> Option<NodeId> {
577        self.info.node_id
578    }
579
580    async fn my_addr(&self) -> Option<NodeAddr> {
581        if let Some(endpoint) = self.node.endpoint() {
582            Some(node_addr(endpoint).await)
583        } else {
584            None
585        }
586    }
587}

621impl<D: SetupData> Builder<D> {
622    /// Creates a new simulation builder with a setup function for setup data.
623    ///
624    /// The setup function is called once before the simulation starts to
625    /// initialize the setup data that will be shared across all nodes.
626    ///
627    /// The setup function can return any type that implements [`SetupData`],
628    /// which is an auto-implemented supertrait for all types that are
629    /// serializable, cloneable, and thread-safe. See [`SetupData`] for details.
630    ///
631    /// # Errors
632    ///
633    /// The setup function should return an error if initialization fails.
634    pub fn with_setup<F, Fut>(setup_fn: F) -> Builder<D>
635    where
636        F: 'static + Send + Sync + FnOnce() -> Fut,
637        Fut: 'static + Send + Future<Output = Result<D>>,
638    {
639        let setup_fn: BoxedSetupFn<D> = Box::new(move || Box::pin(setup_fn()));
640        Builder {
641            spawners: Vec::new(),
642            setup_fn,
643            rounds: 0,
644        }
645    }
646
647    /// Sets the number of rounds this simulation will run.
648    pub fn rounds(mut self, rounds: u32) -> Self {
649        self.rounds = rounds;
650        self
651    }
652
653    /// Adds a group of nodes to spawn in this simulation.
654    ///
655    /// Each node will be created using the provided node builder configuration.
656    ///
657    /// You can create a [`NodeBuilder`] from any type that implements [`Spawn<D>`] where
658    /// `D` is the type returned from [`Self::with_setup`]. If you are not using the setup
659    /// step, `D` defaults to the unit type `()`.
660    pub fn spawn<N: Spawn<D>>(
661        mut self,
662        node_count: u32,
663        node_builder: impl Into<NodeBuilder<N, D>>,
664    ) -> Self {
665        let node_builder = node_builder.into();
666        self.spawners.push((node_count, node_builder.erase()));
667        self
668    }
669
670    /// Builds the final simulation from this configuration.
671    ///
672    /// This method initializes tracing, runs the setup function, and prepares
673    /// all nodes for execution based on the current run mode.
674    ///
675    /// # Errors
676    ///
677    /// Returns an error if setup fails, tracing initialization fails, or
678    /// the configuration is invalid for the current run mode.
679    pub async fn build(self, name: &str) -> Result<Simulation<D>> {
680        let client = TraceClient::from_env_or_local()?;
681        let run_mode = RunMode::from_env()?;
682
683        debug!(%name, ?run_mode, "build simulation run");
684
685        let (trace_id, setup_data) = if matches!(run_mode, RunMode::InitOnly | RunMode::Integrated)
686        {
687            let setup_data = (self.setup_fn)().await?;
688            let encoded_setup_data = Bytes::from(postcard::to_stdvec(&setup_data)?);
689            let node_count = self.spawners.iter().map(|(count, _spawner)| count).sum();
690            let trace_id = client
691                .init_trace(
692                    TraceInfo {
693                        name: name.to_string(),
694                        node_count,
695                        expected_checkpoints: Some(self.rounds as u64),
696                    },
697                    Some(encoded_setup_data),
698                )
699                .await?;
700            info!(%name, node_count, %trace_id, "init simulation");
701
702            (trace_id, setup_data)
703        } else {
704            let info = client.get_trace(name.to_string()).await?;
705            let GetTraceResponse {
706                trace_id,
707                info,
708                setup_data,
709            } = info;
710            info!(%name, node_count=info.node_count, %trace_id, "get simulation");
711            let setup_data = setup_data.context("expected setup data to be set")?;
712            let setup_data: D =
713                postcard::from_bytes(&setup_data).context("failed to decode setup data")?;
714            (trace_id, setup_data)
715        };
716
717        let mut spawners = self
718            .spawners
719            .into_iter()
720            .flat_map(|(count, spawner)| (0..count).map(move |_| spawner.clone()))
721            .enumerate()
722            .map(|(idx, spawner)| (idx as u32, spawner));
723
724        let spawners: Vec<_> = match run_mode {
725            RunMode::InitOnly => vec![],
726            RunMode::Integrated => spawners.collect(),
727            RunMode::Isolated(idx) => vec![
728                spawners
729                    .nth(idx as usize)
730                    .context("invalid isolated index")?,
731            ],
732        };
733
734        Ok(Simulation {
735            run_mode,
736            max_rounds: self.rounds,
737            node_builders: spawners,
738            client,
739            trace_id,
740            setup_data,
741        })
742    }
743}

745/// A configured simulation ready to run.
746///
747/// Contains all the necessary components including the setup data, node spawners,
748/// and tracing client to execute a simulation run.
749pub struct Simulation<D> {
750    trace_id: Uuid,
751    run_mode: RunMode,
752    client: TraceClient,
753    setup_data: D,
754    max_rounds: u32,
755    node_builders: Vec<(u32, ErasedNodeBuilder<D>)>,
756}
757
758impl<D: SetupData> Simulation<D> {
759    /// Runs this simulation to completion.
760    ///
761    /// Spawns all configured nodes concurrently and executes the specified
762    /// number of simulation rounds.
763    ///
764    /// # Errors
765    ///
766    /// Returns an error if any node fails to spawn or if any round fails to execute.
767    pub async fn run(self) -> Result<()> {
768        let cancel_token = CancellationToken::new();
769
770        // Spawn a task to submit logs.
771        let logs_task = {
772            let cancel_token = cancel_token.clone();
773            let client = self.client.clone();
774            let trace_id = self.trace_id;
775            let scope = match self.run_mode {
776                RunMode::Isolated(idx) => Scope::Isolated(idx),
777                _ => Scope::Integrated,
778            };
779            tokio::task::spawn(async move {
780                loop {
781                    let _ = cancel_token
782                        .run_until_cancelled(tokio::time::sleep(Duration::from_secs(1)))
783                        .await;
784                    let lines = self::trace::get_logs().await;
785                    if client.put_logs(trace_id, scope, lines).await.is_err() {
786                        break;
787                    }
788                    if cancel_token.is_cancelled() {
789                        break;
790                    }
791                }
792            })
793        };
794
795        // Spawn and run all nodes concurrently.
796        let result = self
797            .node_builders
798            .into_iter()
799            .map(async |(idx, builder)| {
800                let span = error_span!("sim-node", idx = idx);
801                SimNode::spawn_and_run(
802                    builder,
803                    self.client.clone(),
804                    self.trace_id,
805                    idx,
806                    &self.setup_data,
807                    self.max_rounds,
808                )
809                .instrument(span)
810                .await
811            })
812            .try_join_all()
813            .await
814            .map(|_list| ());
815
816        cancel_token.cancel();
817        logs_task.await?;
818
819        if matches!(self.run_mode, RunMode::InitOnly | RunMode::Integrated) {
820            self.client
821                .close_trace(self.trace_id, to_str_err(&result))
822                .await?;
823        }
824
825        result
826    }
827}
828
829#[derive(Debug, Copy, Clone)]
830enum RunMode {
831    InitOnly,
832    Integrated,
833    Isolated(u32),
834}
835
836impl RunMode {
837    fn from_env() -> Result<Self> {
838        if std::env::var(ENV_TRACE_INIT_ONLY).is_ok() {
839            Ok(Self::InitOnly)
840        } else {
841            match std::env::var(ENV_TRACE_ISOLATED) {
842                Err(_) => Ok(Self::Integrated),
843                Ok(s) => {
844                    let idx = s.parse().with_context(|| {
845                        format!("Failed to parse env var `{ENV_TRACE_ISOLATED}` as number")
846                    })?;
847                    Ok(Self::Isolated(idx))
848                }
849            }
850        }
851    }
852}

854static PERMIT: Semaphore = Semaphore::const_new(1);
855
856/// Runs a simulation function with proper setup and cleanup.
857///
858/// This function handles tracing initialization, sequential execution (via semaphore),
859/// log management, and error reporting for simulation functions.
860///
861/// # Errors
862///
863/// Returns an error if the simulation function fails, the builder fails,
864/// or the simulation execution fails.
865#[doc(hidden)]
866pub async fn run_sim_fn<F, Fut, D, E>(name: &str, sim_fn: F) -> anyhow::Result<()>
867where
868    F: Fn() -> Fut,
869    Fut: Future<Output = Result<Builder<D>, E>>,
870    D: SetupData,
871    anyhow::Error: From<E>,
872{
873    // Ensure simulations run sequentially so that we can extract logs properly.
874    let permit = PERMIT.acquire().await.expect("semaphore closed");
875
876    // Init the global tracing subscriber.
877    self::trace::init();
878    // Clear remaining logs from previous runs.
879    self::trace::global_writer().clear();
880
881    eprintln!("running simulation: {name}");
882    let result = sim_fn()
883        .await
884        .map_err(anyhow::Error::from)
885        .with_context(|| format!("simulation builder function `{name}` failed"))?
886        .build(name)
887        .await
888        .with_context(|| format!("simulation `{name}` failed to build"))?
889        .run()
890        .await
891        .with_context(|| format!("simulation `{name}` failed to run"));
892
893    match &result {
894        Ok(()) => eprintln!("simulation `{name}` passed"),
895        Err(err) => eprintln!("simulation `{name}` failed: {err:#}"),
896    };
897
898    drop(permit);
899
900    result
901}
902
903fn to_str_err<T, E: std::fmt::Display>(res: &Result<T, E>) -> Result<(), String> {
904    if let Some(err) = res.as_ref().err() {
905        Err(err.to_string())
906    } else {
907        Ok(())
908    }
