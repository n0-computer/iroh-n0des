264#[derive(Clone)]
265struct ErasedNodeBuilder<D> {
266    spawn_fn: BoxedSpawnFn<D>,
267    round_fn: BoxedRoundFn<D>,
268    check_fn: Option<BoxedCheckFn<D>>,
269}
270
271impl<N: Spawn<D>, D: UserData> NodeBuilder<N, D> {
272    pub fn new(
273        round_fn: impl for<'a> AsyncCallback<'a, N, RoundContext<'a, D>, Result<bool>>,
274    ) -> Self {
275        let spawn_fn: BoxedSpawnFn<D> = Arc::new(N::spawn_dyn);
276        let round_fn: BoxedRoundFn<D> = Arc::new(move |node, context| {
277            let node = node
278                .as_any_mut()
279                .downcast_mut::<N>()
280                .expect("unreachable: type is statically guaranteed");
281            Box::pin(round_fn(node, context))
282        });
283        Self {
284            phantom: PhantomData,
285            spawn_fn,
286            round_fn,
287            check_fn: None,
288        }
289    }
290
291    pub fn check(
292        mut self,
293        check_fn: impl 'static + for<'a> Fn(&'a N, &RoundContext<'a, D>) -> Result<()>,
294    ) -> Self {
295        let check_fn: BoxedCheckFn<D> = Arc::new(move |node, context| {
296            let node = node
297                .as_any()
298                .downcast_ref::<N>()
299                .expect("unreachable: type is statically guaranteed");
300            check_fn(node, context)
301        });
302        self.check_fn = Some(check_fn);
303        self
304    }
305
306    fn erase(self) -> ErasedNodeBuilder<D> {
307        ErasedNodeBuilder {
308            spawn_fn: self.spawn_fn,
309            round_fn: self.round_fn,
310            check_fn: self.check_fn,
311        }
312    }

345    async fn spawn(builder: ErasedNodeBuilder<D>, node_index: u32, data: &D) -> Result<Self> {
346        info!(idx = node_index, "run start");
347        let mut registry = Registry::default();
348        let mut context = SpawnContext {
349            data,
350            node_index,
351            secret_key: SecretKey::generate(&mut rand::rngs::OsRng),
352            registry: &mut registry,
353        };
354        let node_id = context.node_id();
355        let node = (builder.spawn_fn)(&mut context).await?;
356        if let Some(endpoint) = node.endpoint() {
357            registry.register_all(endpoint.metrics());
358        }
359
360        let info = NodeInfo {
361            node_id,
362            idx: node_index,
363            label: None,
364        };
365        Ok(Self {
366            node,
367            idx: node_index,
368            info,
369            round: 0,
370            round_fn: builder.round_fn,
371            check_fn: builder.check_fn,
372            metrics_encoder: Encoder::new(Arc::new(RwLock::new(registry))),
373            all_nodes: Default::default(),
374        })
375    }
376
377    async fn run(&mut self, client: &ActiveTrace, data: &D, rounds: u32) -> Result<()> {
378        info!(idx = self.idx, "run");
379
380        let info = NodeInfoWithAddr {
381            addr: self.my_addr().await,
382            info: self.info.clone(),
383        };
384        let all_nodes = client.wait_start(info).await?;
385        self.all_nodes = all_nodes;
386
387        let result = self.run_rounds(client, data, rounds).await;
388
389        client.end_trace(to_str_err(&result)).await.ok();
390
391        result
392    }
393
394    async fn run_rounds(&mut self, client: &ActiveTrace, data: &D, rounds: u32) -> Result<()> {
395        while self.round < rounds {
396            self.run_round(client, data)
397                .await
398                .with_context(|| format!("failed at round {}", self.round))?;
399            self.round += 1;
400        }
401        Ok(())
402    }
403
404    async fn run_round(&mut self, client: &ActiveTrace, data: &D) -> Result<()> {
405        info!(idx = self.idx, "run");
406        let context = RoundContext {
407            round: self.round,
408            node_index: self.idx,
409            data,
410            all_nodes: &self.all_nodes,
411        };
412
413        let result = (self.round_fn)(&mut self.node, &context).await;
414
415        let checkpoint = (context.round + 1) as u64;
416        let label = format!("Round {} end", context.round);
417        client
418            .put_checkpoint(checkpoint, self.idx, Some(label), to_str_err(&result))
419            .await?;
420        client
421            .put_metrics(
422                self.node_id(),
423                Some(checkpoint),
424                self.metrics_encoder.export(),
425            )
426            .await?;
427
428        client.wait_checkpoint(checkpoint).await?;
429
430        if let Some(check_fn) = self.check_fn.as_ref() {
431            (check_fn)(&self.node, &context).with_context(|| "check function failed")?;
432        }
433
434        result.map(|_| ())
435    }
436
437    fn node_id(&self) -> NodeId {
438        self.info.node_id
439    }
440
441    async fn my_addr(&self) -> NodeAddr {
442        let mut addr = NodeAddr::new(self.node_id());
443        if let Some(endpoint) = self.node.endpoint() {
444            endpoint.direct_addresses().initialized().await;
445            endpoint.home_relay().initialized().await;
446            addr = endpoint.node_addr().initialized().await;
447        }
448        addr
449    }
450}

499    pub async fn build(self, name: &str) -> Result<Simulation<D>> {
500        let client = TraceClient::from_env_or_local()?;
501        let run_mode = RunMode::from_env()?;
502        let name = name.to_string();
503
504        let node_count = self.spawners.iter().map(|(count, _spawner)| count).sum();
505
506        let data = (self.setup_fn)().await?;
507
508        if matches!(run_mode, RunMode::InitOnly | RunMode::Integrated) {
509            let user_data = Bytes::from(postcard::to_stdvec(&data)?);
510            client
511                .init_trace(
512                    TraceInfo {
513                        name: name.clone(),
514                        node_count,
515                        expected_checkpoints: Some(self.rounds as u64),
516                    },
517                    Some(user_data),
518                )
519                .await?;
520        }
521
522        let mut spawners = self
523            .spawners
524            .into_iter()
525            .flat_map(|(count, spawner)| (0..count).map(move |_| spawner.clone()))
526            .enumerate()
527            .map(|(idx, spawner)| (idx as u32, spawner));
528
529        let spawners: Vec<_> = match run_mode {
530            RunMode::InitOnly => vec![],
531            RunMode::Integrated => spawners.collect(),
532            RunMode::Isolated(idx) => vec![
533                spawners
534                    .nth(idx as usize)
535                    .context("invalid isolated index")?,
536            ],
537        };
538
539        Ok(Simulation {
540            name: name.to_string(),
541            data,
542            max_rounds: self.rounds,
543            spawners,
544            client,
545        })
546    }

549pub struct Simulation<D> {
550    name: String,
551    data: D,
552    max_rounds: u32,
553    spawners: Vec<(u32, ErasedNodeBuilder<D>)>,
554    client: TraceClient,
555}
556
557impl<D: UserData> Simulation<D> {
558    pub async fn run(self) -> Result<()> {
559        // Spawn all nodes concurrently.
560        let mut nodes: Vec<_> = self
561            .spawners
562            .into_iter()
563            .map(async |(idx, spawner)| SimNode::spawn(spawner, idx, &self.data).await)
564            .try_join_all()
565            .await?;
566
567        // Run all nodes concurrently.
568        nodes
569            .iter_mut()
570            .map(async |node| {
571                let idx = node.idx;
572                let scope = ScopeInfo::Isolated(node.info.clone());
573                let (client, data) = self.client.start_trace(self.name.clone(), scope).await?;
574                let data = data.context("expected user data to be set, but wasn't")?;
575                let data: D = postcard::from_bytes(&data).context("failed to decode user data")?;
576                node.run(&client, &data, self.max_rounds)
577                    .await
578                    .with_context(|| format!("node {idx} failed"))
579            })
580            .try_join_all()
581            .await?;
582
583        Ok(())
584    }
585}

587#[derive(Debug, Copy, Clone)]
588enum RunMode {
589    InitOnly,
590    Integrated,
591    Isolated(u32),
592}
593
594impl RunMode {
595    fn from_env() -> Result<Self> {
596        if std::env::var(ENV_TRACE_INIT_ONLY).is_ok() {
597            Ok(Self::InitOnly)
598        } else {
599            match std::env::var(ENV_TRACE_ISOLATED) {
600                Err(_) => Ok(Self::Integrated),
601                Ok(s) => {
602                    let idx = s.parse().with_context(|| {
603                        format!("Failed to parse env var `{ENV_TRACE_ISOLATED}` as number")
604                    })?;
605                    Ok(Self::Isolated(idx))
606                }
607            }
608        }
609    }
610}

612static PERMIT: Semaphore = Semaphore::const_new(1);

614#[doc(hidden)]
615pub async fn run_sim_fn<F, Fut, D, E>(name: &str, sim_fn: F) -> anyhow::Result<()>
616where
617    F: Fn() -> Fut,
618    Fut: Future<Output = Result<Builder<D>, E>>,
619    D: UserData,
620    anyhow::Error: From<E>,
621{
622    // Ensure simulations run sequentially so that we can extract logs properly.
623    let permit = PERMIT.acquire().await.expect("semaphore closed");
624
625    // Init the global tracing subscriber.
626    self::trace::init();
627    // Clear remaining logs from previous runs.
628    self::trace::global_writer().clear();
629
630    eprintln!("running simulation: {name}");
631    let result = sim_fn()
632        .await
633        .map_err(anyhow::Error::from)
634        .with_context(|| format!("simulation builder function `{name}` failed"))?
635        .build(name)
636        .await
637        .with_context(|| format!("simulation `{name}` failed to build"))?
638        .run()
639        .await;
640
641    match &result {
642        Ok(()) => eprintln!("simulation `{name}` passed"),
643        Err(err) => eprintln!("simulation `{name}` failed: {err:#}"),
644    };
645    result.with_context(|| format!("simulation `{name}` failed to run"))?;
646
647    drop(permit);
648    Ok(())
649}
