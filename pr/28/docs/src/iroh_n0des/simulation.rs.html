119impl<N: N0de> Simulation<N> {
120    pub fn builder<F>(round: F) -> SimulationBuilder<N>
121    where
122        F: for<'a> AsyncCallback<'a, N, bool>,
123    {
124        let round_fn: BoxedRoundFn<N> =
125            Arc::new(move |context: &Context, node: &mut N| Box::pin(round(context, node)));
126        SimulationBuilder::<N> {
127            max_rounds: 100,
128            node_count: 2,
129            round_fn,
130            check_fn: None,
131            run_mode: RunMode::Integrated,
132            _node: PhantomData,
133        }
134    }
135
136    fn context(&self, node_index: usize) -> Context {
137        Context {
138            addrs: self.addrs.clone(),
139            node_index,
140            round: self.round,
141        }
142    }
143    pub async fn run(&mut self) -> Result<()> {
144        let res = self.run_inner().await;
145        if let Some(ref client) = self.trace_client {
146            let rpc_res = res.as_ref().map(|_| ()).map_err(|e| e.to_string());
147            client.end_trace(rpc_res).await?;
148        }
149        res
150    }
151
152    async fn if_isolated<R>(
153        &self,
154        f: impl AsyncFnOnce(&ActiveTrace) -> Result<R>,
155    ) -> Result<Option<R>> {
156        if let Some(ref client) = self.trace_client {
157            if let RunMode::Isolated(_) = self.run_mode {
158                return Ok(Some(f(client).await?));
159            }
160        }
161        Ok(None)
162    }
163
164    async fn run_inner(&mut self) -> Result<()> {
165        self.if_isolated(async |client| {
166            client.wait_checkpoint(0, self.node_count).await?;
167            Ok(())
168        })
169        .await?;
170
171        while self.round < self.max_rounds {
172            if let Err(err) = self.run_round().await {
173                error!("Simulation failed at round {}: {err:#}", self.round);
174                return Err(err);
175            }
176            self.round += 1;
177        }
178        Ok(())
179    }
180
181    async fn run_round(&mut self) -> Result<()> {
182        println!("Round {}", self.round);
183
184        let ctx = self.context(0);
185        let round_fn = self.round_fn.clone();
186
187        let run_fn = async |node: &mut SimNode<N>| {
188            let mut ctx = ctx.clone();
189            ctx.node_index = node.idx;
190            let res = (round_fn)(&ctx, &mut node.node)
191                .await
192                .with_context(|| format!("Node {} failed in round {}", ctx.node_index, ctx.round));
193
194            if let Err(err) = res.as_ref() {
195                error!("{err:#}");
196            }
197
198            if let Some(ref client) = self.trace_client {
199                let checkpoint = ctx.round + 1;
200                client
201                    .put_checkpoint(
202                        checkpoint,
203                        ctx.node_index as u32,
204                        Some(format!("round {checkpoint} end")),
205                        res.as_ref().map_err(|err| format!("{err:#}")).map(|_| ()),
206                    )
207                    .await?;
208
209                let metrics = node.encoder.export();
210                client
211                    .put_metrics(node.node_id(), Some(checkpoint), metrics)
212                    .await?;
213            }
214
215            anyhow::Ok(())
216        };
217        let futures = self.nodes.iter_mut().map(|node| {
218            let span = node.span.clone();
219            run_fn(node).instrument(span)
220        });
221
222        let _res: Vec<_> = n0_future::FuturesOrdered::from_iter(futures)
223            .try_collect()
224            .await?;
225
226        self.if_isolated(async |client| {
227            client
228                .wait_checkpoint(self.round + 1, self.node_count)
229                .await?;
230            Ok(())
231        })
232        .await?;
233
234        println!("Round {} completed", self.round);
235        if let Some(check) = &self.check_fn {
236            for node in self.nodes.iter() {
237                let _guard = node.span.enter();
238                let ctx = self.context(node.idx);
239                (check)(&ctx, &node.node)?;
240            }
241        }
242
243        if let Some(ref client) = self.trace_client {
244            submit_logs(client).await?;
245        }
246
247        Ok(())
248    }
249}

287    pub async fn build(self, trace_name: &str) -> Result<Simulation<N>> {
288        let nodes: Vec<_> = match self.run_mode {
289            RunMode::Integrated => {
290                FuturesUnordered::from_iter(
291                    (0usize..self.node_count as usize).map(|i| SimNode::<N>::spawn(i)),
292                )
293                .try_collect()
294                .await?
295            }
296            RunMode::Isolated(idx) => {
297                anyhow::ensure!(
298                    idx < self.node_count,
299                    "Isolated index must be lower than max node count"
300                );
301                vec![SimNode::<N>::spawn(idx as usize).await?]
302            }
303        };
304
305        let client = if let Some(client) = client_from_env()? {
306            if matches!(self.run_mode, RunMode::Integrated) {
307                let info = TraceInfo {
308                    name: trace_name.to_string(),
309                    expected_nodes: Some(self.node_count),
310                    expected_checkpoints: Some(self.max_rounds),
311                };
312                client.init_trace(info).await?;
313            }
314            let scope = match self.run_mode {
315                RunMode::Integrated => {
316                    let node_info = nodes.iter().map(|node| node.info()).collect();
317                    ScopeInfo::Integrated(node_info)
318                }
319                RunMode::Isolated(idx) => {
320                    let node_info = nodes[0].info();
321                    debug_assert_eq!(node_info.idx, idx);
322                    ScopeInfo::Isolated(node_info)
323                }
324            };
325            let trace_client = client
326                .start_trace(trace_name.to_string(), scope)
327                .await
328                .context("failed to connect to sim server")?;
329            Some(trace_client)
330        } else {
331            None
332        };
333
334        let addrs = match (&client, self.run_mode) {
335            (Some(client), RunMode::Isolated(_)) => {
336                let node = &nodes[0];
337                let info = node.info_with_addr().await?;
338
339                client.wait_start(info, self.node_count).await?
340            }
341            _ => {
342                let addrs: Vec<_> = FuturesUnordered::from_iter(nodes.iter().map(|n| async {
343                    let info = n.info();
344                    let addr = node_addr(&n.endpoint).await?;
345                    anyhow::Ok(NodeInfoWithAddr { info, addr })
346                }))
347                .try_collect()
348                .await?;
349                addrs
350            }
351        };
352
353        Ok(Simulation {
354            name: trace_name.to_string(),
355            max_rounds: self.max_rounds,
356            round_fn: self.round_fn,
357            check_fn: self.check_fn,
358            nodes,
359            addrs: Arc::new(addrs.into_iter().map(|n| (n.info.idx, n)).collect()),
360            trace_client: client,
361            round: 0,
362            run_mode: self.run_mode,
363            node_count: self.node_count,
364        })
365    }

378impl<N: N0de> SimNode<N> {
379    async fn spawn(idx: usize) -> Result<Self> {
380        let span = error_span!("sim-node", idx, round = tracing::field::Empty);
381        let start = n0_future::time::Instant::now();
382        let (node, endpoint, registry) = async move {
383            let mut registry = Registry::default();
384            let endpoint = Endpoint::builder().bind().await?;
385
386            // TODO(Frando): enable metrics support for iroh 0.35
387            // iroh@0.35 uses iroh-metrics@0.34, which is a major version down from iroh-metrics@0.35
388            // which is used here, thus the traits are incompatible.
389            // Add shim to iroh-metrics@0.35 to support registering metrics from iroh-metrics@0.34
390            #[cfg(not(feature = "iroh_v035"))]
391            registry.register_all(endpoint.metrics());
392
393            let node = N::spawn(endpoint.clone(), &mut registry).await?;
394            anyhow::Ok((node, endpoint, registry))
395        }
396        .instrument(span.clone())
397        .await?;
398        Ok(Self {
399            idx,
400            node,
401            endpoint,
402            encoder: Encoder::new(Arc::new(RwLock::new(registry))),
403            span,
404            spawn_time: start.elapsed(),
405        })
406    }

453static PERMIT: Semaphore = Semaphore::const_new(1);
454
455#[doc(hidden)]
456pub async fn run_sim_fn<F, Fut, N, E>(name: &str, sim_fn: F) -> anyhow::Result<()>
457where
458    F: Fn() -> Fut,
459    Fut: Future<Output = Result<SimulationBuilder<N>, E>>,
460    N: N0de,
461    E: Into<anyhow::Error>,
462{
463    // ensure simulations run sequentially so that we can extract logs properly.
464    let permit = PERMIT.acquire().await.unwrap();
465
466    if is_sim_env() {
467        tracing::debug!("running in simulation env: ensure tracing subscriber is setup");
468        self::trace::try_init_global_subscriber();
469        self::trace::global_writer().clear();
470    }
471
472    let isolated: Option<u32> = match std::env::var(ENV_TRACE_ISOLATED) {
473        Err(_) => None,
474        Ok(s) => Some(s.parse().with_context(|| {
475            format!("Failed to parse env var `{ENV_TRACE_ISOLATED}` as number")
476        })?),
477    };
478
479    println!("running simulation: {name}");
480    let mut builder = sim_fn()
481        .await
482        .map_err(|err| {
483            // Not sure why, but anyhow::Error::from doesn't work here.
484            let err: anyhow::Error = err.into();
485            err
486        })
487        .with_context(|| format!("simulation builder function `{name}` failed"))?;
488
489    if std::env::var(ENV_TRACE_INIT_ONLY).is_ok() {
490        let client = client_from_env()?.expect("failed to build client");
491        client
492            .init_trace(TraceInfo {
493                name: name.to_string(),
494                expected_nodes: Some(builder.node_count),
495                expected_checkpoints: Some(builder.max_rounds),
496            })
497            .await?;
498        println!("{}", builder.node_count);
499        return Ok(());
500    }
501
502    if let Some(isolated_idx) = isolated {
503        builder = builder.isolated(isolated_idx);
504    }
505
506    let mut sim = builder
507        .build(name)
508        .await
509        .with_context(|| format!("simulation `{name}` failed to build"))?;
510    let res = sim.run().await;
511    match &res {
512        Ok(()) => println!("simulation passed"),
513        Err(err) => println!("simulation failed: {err:#}"),
514    };
515    res.with_context(|| format!("simulation `{name}` failed to run"))?;
516
517    drop(permit);
518    Ok(())
519}

732    #[crate::sim]
733    async fn custom_events() -> Result<SimulationBuilder<NoopNode>> {
734        async fn tick(ctx: &Context, node: &mut NoopNode) -> Result<bool> {
735            let me = node.endpoint.node_id();
736            if let Some((id, other_node)) = node.active_event.take() {
737                self::events::event_end(me, other_node, "done", id);
738            }
739            let other = ctx
740                .all_other_nodes(me)
741                .choose(&mut rand::thread_rng())
742                .unwrap();
743            let id = format!("ping:n{}:r{}", ctx.node_index, ctx.round);
744            self::events::event_start(
745                me.fmt_short(),
746                other.node_id.fmt_short(),
747                format!("ping round {}", ctx.round),
748                Some(&id),
749            );
750            node.active_event = Some((id, other.node_id));
751            Ok(true)
752        }
753
754        Ok(Simulation::builder(tick).max_rounds(4).node_count(4))
755    }
